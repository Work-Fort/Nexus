// Package client provides a typed Go HTTP client for the Nexus daemon API.
package client

import (
	"errors"
	"fmt"
)

// Sentinel errors mapped from HTTP status codes.
var (
	ErrNotFound = errors.New("not found")
	ErrConflict = errors.New("conflict")
	ErrBadRequest = errors.New("bad request")
)

// APIError represents an error response from the Nexus API.
type APIError struct {
	StatusCode int    `json:"status"`
	Title      string `json:"title"`
	Detail     string `json:"detail,omitempty"`
	wrapped    error
}

func (e *APIError) Error() string {
	if e.Detail != "" {
		return fmt.Sprintf("%d: %s: %s", e.StatusCode, e.Title, e.Detail)
	}
	return fmt.Sprintf("%d: %s", e.StatusCode, e.Title)
}

func (e *APIError) Unwrap() error { return e.wrapped }

// SPDX-License-Identifier: GPL-3.0-or-later
package client

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"time"
)

// Client is a typed HTTP client for the Nexus daemon API.
type Client struct {
	base string
	http *http.Client
}

// New creates a new Nexus API client. baseURL is the daemon address,
// e.g. "http://localhost:9600".
func New(baseURL string) *Client {
	return &Client{
		base: baseURL,
		http: &http.Client{Timeout: 60 * time.Second},
	}
}

func (c *Client) get(ctx context.Context, path string) (*http.Response, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, c.base+path, nil)
	if err != nil {
		return nil, err
	}
	return c.http.Do(req)
}

func (c *Client) post(ctx context.Context, path string, body any) (*http.Response, error) {
	return c.doJSON(ctx, http.MethodPost, path, body)
}

func (c *Client) put(ctx context.Context, path string, body any) (*http.Response, error) {
	return c.doJSON(ctx, http.MethodPut, path, body)
}

func (c *Client) patch(ctx context.Context, path string, body any) (*http.Response, error) {
	return c.doJSON(ctx, http.MethodPatch, path, body)
}

func (c *Client) del(ctx context.Context, path string) (*http.Response, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodDelete, c.base+path, nil)
	if err != nil {
		return nil, err
	}
	return c.http.Do(req)
}

func (c *Client) doJSON(ctx context.Context, method, path string, body any) (*http.Response, error) {
	var r io.Reader
	if body != nil {
		buf, err := json.Marshal(body)
		if err != nil {
			return nil, fmt.Errorf("marshal body: %w", err)
		}
		r = bytes.NewReader(buf)
	}
	req, err := http.NewRequestWithContext(ctx, method, c.base+path, r)
	if err != nil {
		return nil, err
	}
	if body != nil {
		req.Header.Set("Content-Type", "application/json")
	}
	return c.http.Do(req)
}

func checkStatus(resp *http.Response, expected int) error {
	if resp.StatusCode == expected {
		return nil
	}
	var apiErr APIError
	if err := json.NewDecoder(resp.Body).Decode(&apiErr); err != nil {
		return &APIError{StatusCode: resp.StatusCode, Title: fmt.Sprintf("unexpected status %d", resp.StatusCode)}
	}
	apiErr.StatusCode = resp.StatusCode
	switch resp.StatusCode {
	case http.StatusNotFound:
		apiErr.wrapped = ErrNotFound
	case http.StatusConflict:
		apiErr.wrapped = ErrConflict
	case http.StatusBadRequest:
		apiErr.wrapped = ErrBadRequest
	}
	return &apiErr
}

func decodeJSON[T any](resp *http.Response) (*T, error) {
	var v T
	if err := json.NewDecoder(resp.Body).Decode(&v); err != nil {
		return nil, fmt.Errorf("decode response: %w", err)
	}
	return &v, nil
}

// VM represents a virtual machine returned by the API.
type VM struct {
	ID              string  `json:"id"`
	Name            string  `json:"name"`
	Role            string  `json:"role"`
	State           string  `json:"state"`
	Image           string  `json:"image"`
	Runtime         string  `json:"runtime"`
	IP              string  `json:"ip,omitempty"`
	Gateway         string  `json:"gateway,omitempty"`
	RootSize        *string `json:"root_size,omitempty"`
	RestartPolicy   string  `json:"restart_policy"`
	RestartStrategy string  `json:"restart_strategy"`
	CreatedAt       string  `json:"created_at"`
	StartedAt       *string `json:"started_at,omitempty"`
	StoppedAt       *string `json:"stopped_at,omitempty"`
}

// CreateVMParams holds parameters for creating a VM.
type CreateVMParams struct {
	Name            string `json:"name"`
	Role            string `json:"role"`
	Image           string `json:"image,omitempty"`
	Runtime         string `json:"runtime,omitempty"`
	RootSize        string `json:"root_size,omitempty"`
	RestartPolicy   string `json:"restart_policy,omitempty"`
	RestartStrategy string `json:"restart_strategy,omitempty"`
}

// ExecResult holds command output from a VM.
type ExecResult struct {
	ExitCode int    `json:"exit_code"`
	Stdout   string `json:"stdout"`
	Stderr   string `json:"stderr"`
}

// ListVMsFilter constrains VM listing.
type ListVMsFilter struct {
	Role string
}

func (c *Client) CreateVM(ctx context.Context, params CreateVMParams) (*VM, error) {
	resp, err := c.post(ctx, "/v1/vms", params)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if err := checkStatus(resp, http.StatusCreated); err != nil {
		return nil, err
	}
	return decodeJSON[VM](resp)
}

func (c *Client) ListVMs(ctx context.Context, filter ListVMsFilter) ([]*VM, error) {
	path := "/v1/vms"
	if filter.Role != "" {
		path += "?role=" + filter.Role

	resp, err := c.get(ctx, path)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if err := checkStatus(resp, http.StatusOK); err != nil {
		return nil, err
	}
	return decodeJSON[[]*VM](resp)
}

import (
	"context"

	"io"
	"net/http"

// ImportResult holds the result of importing a VM backup.
type ImportResult struct {
	VM       VM       `json:"vm"`
	Warnings []string `json:"warnings,omitempty"`
}

// ExportVM streams a VM backup archive to the provided writer.
func (c *Client) ExportVM(ctx context.Context, ref string, includeDevices bool, w io.Writer) error {
	path := fmt.Sprintf("/v1/vms/%s/export?include_devices=%t", ref, includeDevices)
	resp, err := c.post(ctx, path, nil)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	if err := checkStatus(resp, http.StatusOK); err != nil {
		return err
	}
	_, err = io.Copy(w, resp.Body)
	return err
}

// ImportVM imports a VM from a backup archive read from the provided reader.
func (c *Client) ImportVM(ctx context.Context, r io.Reader, strictDevices bool) (*ImportResult, error) {
	path := fmt.Sprintf("/v1/vms/import?strict_devices=%t", strictDevices)
	req, err := http.NewRequestWithContext(ctx, http.MethodPost, c.base+path, r)
	if err != nil {
		return nil, err
	}
	req.Header.Set("Content-Type", "application/zstd")
	resp, err := c.http.Do(req)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
	if err := checkStatus(resp, http.StatusCreated); err != nil {
		return nil, err
	}
	return decodeJSON[ImportResult](resp)
}

	srv := httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		if r.Method != http.MethodGet || r.URL.Path != "/v1/vms" {
			t.Errorf("unexpected request: %s %s", r.Method, r.URL.Path)

	vms, err := c.ListVMs(context.Background(), ListVMsFilter{})
	if err != nil {
		t.Fatalf("ListVMs: %v", err)

	if len(*vms) != 1 {
		t.Fatalf("expected 1 VM, got %d", len(*vms))

package main

import (
	"fmt"

	"github.com/spf13/cobra"
)

func newExecCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "exec <vm> -- <cmd...>",
		Short: "Execute a command in a VM",
		Args:  cobra.MinimumNArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			vmRef := args[0]

			// Find -- separator
			var cmdArgs []string
			dashIdx := cmd.ArgsLenAtDash()
			if dashIdx >= 0 {
				cmdArgs = args[dashIdx:]
			} else if len(args) > 1 {
				cmdArgs = args[1:]
			}
			if len(cmdArgs) == 0 {
				return fmt.Errorf("no command specified (use -- before the command)")
			}

			result, err := apiClient.ExecVM(cmd.Context(), vmRef, cmdArgs)
			if err != nil {
				return err
			}
			if jsonFlag {
				printJSON(result)
				return nil
			}
			if result.Stdout != "" {
				fmt.Print(result.Stdout)
			}
			if result.Stderr != "" {
				fmt.Fprint(cmd.ErrOrStderr(), result.Stderr)
			}
			if result.ExitCode != 0 {
				return fmt.Errorf("exit code %d", result.ExitCode)
			}
			return nil
		},
	}
}

// SPDX-License-Identifier: GPL-3.0-or-later
package client

import (
	"context"
	"strings"

	"nhooyr.io/websocket"
)

// ConsoleOpts configures the console connection.
type ConsoleOpts struct {
	Cmd string // shell command to run (optional)
}

// Console opens a WebSocket connection to the VM's console.
// The caller is responsible for closing the returned connection.
func (c *Client) Console(ctx context.Context, ref string, opts ConsoleOpts) (*websocket.Conn, error) {
	u := c.base + "/v1/vms/" + ref + "/console"
	if opts.Cmd != "" {
		u += "?cmd=" + opts.Cmd
	}

	// Convert http:// to ws:// for WebSocket
	u = strings.Replace(u, "http://", "ws://", 1)
	u = strings.Replace(u, "https://", "wss://", 1)

	conn, _, err := websocket.Dial(ctx, u, nil)
	if err != nil {
		return nil, err
	}
	return conn, nil
}
```

package main

import (
	"context"
	"encoding/json"
	"fmt"
	"io"
	"os"
	"os/signal"
	"syscall"

	"github.com/spf13/cobra"
	"golang.org/x/term"
	"nhooyr.io/websocket"

	"github.com/Work-Fort/Nexus/client"
)

func newConsoleCmd() *cobra.Command {
	var shellCmd string
	cmd := &cobra.Command{
		Use:   "console <vm>",
		Short: "Interactive terminal access to a VM",
		Long:  "Opens a terminal session to a VM via WebSocket. Use ~. to disconnect.",
		Args:  cobra.ExactArgs(1),
		RunE: func(cmd *cobra.Command, args []string) error {
			conn, err := apiClient.Console(cmd.Context(), args[0], client.ConsoleOpts{Cmd: shellCmd})
			if err != nil {
				return fmt.Errorf("connect: %w", err)
			}
			defer conn.CloseNow() //nolint:errcheck

			return runConsole(cmd.Context(), conn)
		},
	}
	cmd.Flags().StringVar(&shellCmd, "cmd", "", "Shell command to run")
	return cmd
}

func runConsole(ctx context.Context, conn *websocket.Conn) error {
	fd := int(os.Stdin.Fd())
	if !term.IsTerminal(fd) {
		return fmt.Errorf("stdin is not a terminal")
	}

	oldState, err := term.MakeRaw(fd)
	if err != nil {
		return fmt.Errorf("set raw mode: %w", err)
	}
	defer term.Restore(fd, oldState) //nolint:errcheck

	ctx, cancel := context.WithCancel(ctx)
	defer cancel()

	// Handle SIGWINCH for terminal resize
	sigCh := make(chan os.Signal, 1)
	signal.Notify(sigCh, syscall.SIGWINCH)
	defer signal.Stop(sigCh)

	// Send initial terminal size
	sendResize(conn, fd)

	go func() {
		for range sigCh {
			sendResize(conn, fd)
		}
	}()

	// Read from WebSocket → stdout
	go func() {
		defer cancel()
		for {
			_, data, err := conn.Read(ctx)
			if err != nil {
				return
			}
			os.Stdout.Write(data)
		}
	}()

	// Read from stdin → WebSocket, with ~. escape detection
	var lastWasNewline bool
	var lastWasTilde bool
	buf := make([]byte, 1024)
	for {
		n, err := os.Stdin.Read(buf)
		if err != nil {
			if err == io.EOF {
				return nil
			}
			return err
		}

		for i := 0; i < n; i++ {
			b := buf[i]
			if lastWasTilde && b == '.' {
				fmt.Fprintln(os.Stderr, "\r\nConnection closed.")
				return nil
			}
			lastWasTilde = lastWasNewline && b == '~'
			lastWasNewline = b == '\r' || b == '\n'
		}

		if err := conn.Write(ctx, websocket.MessageText, buf[:n]); err != nil {
			return nil
		}
	}
}

type resizeMsg struct {
	Rows int `json:"rows"`
	Cols int `json:"cols"`
}

func sendResize(conn *websocket.Conn, fd int) {
	w, h, err := term.GetSize(fd)
	if err != nil {
		return
	}
	data, _ := json.Marshal(resizeMsg{Rows: h, Cols: w})
	conn.Write(context.Background(), websocket.MessageText, data) //nolint:errcheck
}

// SPDX-License-Identifier: GPL-3.0-or-later
package main

import (
	"bufio"
	"fmt"
	"io"
	"net/http"
	"os"
	"strings"

	"github.com/spf13/cobra"
	"github.com/spf13/viper"
)

func newMCPBridgeCmd() *cobra.Command {
	return &cobra.Command{
		Use:   "mcp-bridge",
		Short: "Transparent stdio-to-HTTP MCP proxy",
		Long: `Reads newline-delimited JSON-RPC from stdin, POSTs each message
to /mcp on the daemon, and writes the response to stdout.
Runs until stdin EOF. Used by agent VMs for MCP tool access.`,
		RunE: func(cmd *cobra.Command, args []string) error {
			addr := viper.GetString("addr")
			mcpURL := addr + "/mcp"

			scanner := bufio.NewScanner(os.Stdin)
			for scanner.Scan() {
				line := scanner.Text()
				if line == "" {
					continue
				}

				resp, err := http.Post(mcpURL, "application/json", strings.NewReader(line))
				if err != nil {
					fmt.Fprintf(os.Stderr, "mcp-bridge: POST error: %v\n", err)
					continue
				}
				io.Copy(os.Stdout, resp.Body)
				resp.Body.Close()
				fmt.Fprintln(os.Stdout) // newline delimiter
			}

			return scanner.Err()
		},
	}
}
```
