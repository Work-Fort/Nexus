# VM Tags Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Replace the hardcoded `role` enum with free-form string tags on VMs, stored in a join table with AND/OR filtering support.

**Architecture:** New `vm_tags` join table with a goose migration that migrates existing role values. Domain model replaces `Role VMRole` with `Tags []string`. Store loads tags via a second query and stitches them onto VM structs. API changes `role` to `tags` throughout. All consumers (httpapi, tests, backup, E2E harness) update accordingly.

**Tech Stack:** Go, SQLite (goose migrations, sqlc), huma v2 HTTP API

### Task 1: SQLite Migration — Create vm_tags Table and Drop role

**Files:**
- Create: `internal/infra/sqlite/migrations/009_tags.sql`

**Step 1: Create the migration**

```sql
-- SPDX-License-Identifier: Apache-2.0

-- +goose Up
CREATE TABLE vm_tags (
    vm_id TEXT NOT NULL REFERENCES vms(id) ON DELETE CASCADE,
    tag   TEXT NOT NULL,
    PRIMARY KEY (vm_id, tag)
);

CREATE INDEX idx_vm_tags_tag ON vm_tags(tag);

-- Migrate existing role values into vm_tags.
INSERT INTO vm_tags (vm_id, tag) SELECT id, role FROM vms;

-- Drop role column and its index.
DROP INDEX IF EXISTS idx_vms_role;
ALTER TABLE vms DROP COLUMN role;

-- +goose Down
ALTER TABLE vms ADD COLUMN role TEXT NOT NULL DEFAULT 'agent' CHECK (role IN ('agent', 'service'));
UPDATE vms SET role = (SELECT tag FROM vm_tags WHERE vm_tags.vm_id = vms.id LIMIT 1);
CREATE INDEX idx_vms_role ON vms(role);
DROP INDEX IF EXISTS idx_vm_tags_tag;
DROP TABLE IF EXISTS vm_tags;
```

func (s *Store) listByTags(ctx context.Context, tags []string, matchMode string) ([]*domain.VM, error) {
	// Build dynamic IN clause.
	placeholders := make([]string, len(tags))
	args := make([]any, len(tags))
	for i, t := range tags {
		placeholders[i] = "?"
		args[i] = t
	}
	inClause := strings.Join(placeholders, ",")

	var query string
	if matchMode == "any" {
		query = `SELECT DISTINCT v.id, v.name, v.image, v.runtime, v.state,
			v.created_at, v.started_at, v.stopped_at, v.ip, v.gateway, v.netns_path,
			v.dns_servers, v.dns_search, v.root_size, v.restart_policy, v.restart_strategy
			FROM vms v JOIN vm_tags t ON v.id = t.vm_id
			WHERE t.tag IN (` + inClause + `) ORDER BY v.created_at DESC`
	} else {
		query = `SELECT v.id, v.name, v.image, v.runtime, v.state,
			v.created_at, v.started_at, v.stopped_at, v.ip, v.gateway, v.netns_path,
			v.dns_servers, v.dns_search, v.root_size, v.restart_policy, v.restart_strategy
			FROM vms v JOIN vm_tags t ON v.id = t.vm_id
			WHERE t.tag IN (` + inClause + `)
			GROUP BY v.id HAVING COUNT(DISTINCT t.tag) = ?
			ORDER BY v.created_at DESC`
		args = append(args, len(tags))
	}

	rows, err := s.db.QueryContext(ctx, query, args...)
	if err != nil {
		return nil, fmt.Errorf("list vms by tags: %w", err)
	}
	defer rows.Close()

	var vms []*domain.VM
	for rows.Next() {
		var row Vm
		if err := rows.Scan(&row.ID, &row.Name, &row.Image, &row.Runtime,
			&row.State, &row.CreatedAt, &row.StartedAt, &row.StoppedAt,
			&row.Ip, &row.Gateway, &row.NetnsPath, &row.DnsServers,
			&row.DnsSearch, &row.RootSize, &row.RestartPolicy, &row.RestartStrategy); err != nil {
			return nil, fmt.Errorf("scan vm: %w", err)
		}
		vm, err := vmFromRow(row)
		if err != nil {
			return nil, err
		}
		vm.Tags, err = s.loadTags(ctx, vm.ID)
		if err != nil {
			return nil, fmt.Errorf("load tags: %w", err)
		}
		vms = append(vms, vm)
	}
	return vms, rows.Err()
}

func (s *VMService) SetTags(ctx context.Context, ref string, tags []string) (*domain.VM, error) {

func matchTags(vmTags, filterTags []string, mode string) bool {
	tagSet := make(map[string]bool, len(vmTags))
	for _, t := range vmTags {
		tagSet[t] = true
	}
	if mode == "any" {
		for _, t := range filterTags {
			if tagSet[t] {

			}

		return false
	}
	// AND (default)
	for _, t := range filterTags {
		if !tagSet[t] {
			return false
		}
	}
	return true
}
